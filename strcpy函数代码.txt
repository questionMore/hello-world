//自己实现字符串拷贝函数
char* StrCpy(char* src){
	//1. 计算src的长度len.
	int len = 0;
	while(*src)
	len++;
	//2. 用malloc函数申请长度为len+1的内存空间.
	char* result = (char*)malloc(sizeof(char)*(len+1));
	//3. 将src字符串逐个字符复制到新申请的内存空间.
	for(int i = 0; src[i];i++)
	result[i]=src[i];
	//4. 新申请内存空间的最后一位放入'\0'.
	result[len]=0;
	//5. 返回新申请到的内存空间指针.
	return result;
}
